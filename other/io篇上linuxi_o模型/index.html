IO篇上 (Linux I/O模型)

September 14, 2024

网络数据流向方式，以数据接收方来看，从网卡->内核->用户空间，数据经过两次复制（相关资料 零拷贝）

1BIO#

#include <stdio.h>
#include <string.h>
#include <arpa/inet.h>
#include <sys/types.h>
#include <sys/socket.h>

#define MAX_SIZE 512
#define PORT 10007

int main(void)
{
    int sockfd;
    int sock_fd;
    int recvnum;
    int addrlen = sizeof(struct sockaddr);
    struct sockaddr_in my_addr;
    struct sockaddr addr;
    char buf[MAX_SIZE];
    // 填充服务器端的数据，用于套接字绑定
    bzero(&my_addr, sizeof(struct sockaddr_in));
    my_addr.sin_family = AF_INET;   // 设置为IPV4
    my_addr.sin_port = htons(PORT); // 将端口号主机序转换为网络序
    my_addr.sin_addr.s_addr = inet_addr("127.0.0.1");
    // 创建套接字
    sockfd = socket(AF_INET, SOCK_STREAM, 0);
    if (sockfd < 0)
    {
        printf("create socket error!\n");
        return 0;
    }
    // 绑定套接字
    if (bind(sockfd, (struct sockaddr *)&my_addr, addrlen) < 0)
    {
        printf("bind error!\n");
        return 0;
    }
    // 监听端口和ip，设置最大连接数为3
    if (listen(sockfd, 3) < 0)
    {
        printf("listen error!\n");
        return 0;
    }
    // 建立服务器端和客户端连接
    sock_fd = accept(sockfd, &addr, &addrlen);
    // 建立连接后，产生新的套接字
    if (sock_fd < 0)
    {
        printf("accept error!\n");
        return 0;
    }
    // 接收数据
    if ((recvnum = recv(sock_fd, (void *)buf, MAX_SIZE, 0)) < 0)
    {
        printf("recv error!\n");
        return 0;
    }
    buf[recvnum] = '\0';
    printf("recv from client: %s\n", buf);
    memset(buf, 0, MAX_SIZE);
    // 关闭连接
    close(sockfd);
    close(sock_fd);
    return 0;
}

单线程方式，从accept建立连接到recv发起系统调用之后，整个线程在事件完成之前会一直阻塞住。

由下图可见

在回车键连接之前，sock_fd一直没有拿到阻塞在accept函数位置.

下一步，按下回车键，如下图，连接成功，accept函数不阻塞，到下一步

如图，连接了但未发送数据，recv阻塞住

发送数据recv，recv不阻塞，成功将数据放到了数组中，结束。

!

优化方式:多线程方式，每一个请求，单独开一个线程处理，缺点线程占内存，高并发下线程占用内存空间，调度CPU资源大

gateway->（阻塞）kernel space->（阻塞） user space

2 NIO#

#include <sys/types.h>
#include <sys/socket.h>
#include <arpa/inet.h>
#include <fcntl.h>
#include <stdio.h>
#include <errno.h>
#include <string.h>
#include <sys/select.h>
#define MAX_SIZE 512
#define PORT 10007
int main(void)
{

    int sockfd;
    int sock_fd;
    int recvnum;
    int addrlen = sizeof(struct sockaddr);
    struct sockaddr_in my_addr;

    my_addr.sin_addr.s_addr = inet_addr("127.0.0.1");

    sockfd = socket(AF_INET, SOCK_STREAM, 0);

    if (sockfd < 0)
    {
        printf("create socket error!\n");
        return 0;
    }
    // 绑定套接字
    if (bind(sockfd, (struct sockaddr *)&my_addr, addrlen) < 0)
    {
        printf("bind error!\n");

    }

    // 建立服务器端和客户端连接
    sock_fd = accept(sockfd, &addr, &addrlen);
    // 建立连接后，产生新的套接字
    if (sock_fd < 0)
    {
        printf("accept error!\n");
        return 0;
    }
    // if (0 > fcntl(sockfd, F_SETFL, fcntl(sockfd, F_GETFL, 0) | /*O_NDELAY*/ O_NONBLOCK))
    // {

    while (1)
    {
        int len_recv = recv(sock_fd, (void *)buf, MAX_SIZE, MSG_DONTWAIT);
        if (len_recv == -1)
        {
            if (len_recv == EAGAIN || len_recv == EWOULDBLOCK)
            {
                continue;
            }
            continue;

        buf[recvnum] = '\0';

        // 关闭连接
        close(sockfd);
        close(sock_fd);
        return 0;
    }
    return 0;
}

在fcntl设置fd之前，accept同样阻塞，所以fcntl是为了解决accept阻塞问题。

nio解决的问题是recv系统调用之后，线程不会阻塞卡在recv函数上，内核马上返回给线程，如果数据还没准备好，此时会返回一个error。然后循环往复的进行recvform系统调用。直到数据准备好，再拷贝数据到进程，进行数据处理。

如下图，启动成功之后，fd并没有返回，如果通过fcntl设置了fd，会立即返回，这个流程就不截图了。

建立连接，连接成功。

关键点来了,从命令行可见，并没有send数据，但recv返回了fd，成功变成了未阻塞。

send数据fd返回，可以跳出循环了，这里写的有点问题，recv的fd返回了，buf数组也有数据了，结束。

总结：其实non blocking所以没有占用线程了，但实际上线程一直在while 1对cpu资源同样消耗巨大。

3 IO Multiplexin#

select poll epoll都是这么做的

多路复用的思想是多个连结过来共用一个线程，同时对多个IO请求过来进行监听，会返回一个fd，然后内核将fd保存到(数组 链表 链表+红黑树)，轮询这些数据结构，而不是轮询单个fd，这样从而实现一个线程对应多个连接。当其中任何一个连接的数据准好了( 重点！！如果有其他连接也准备好，会进行阻塞)，然后在进行recv系统调用，就能返回数据到buf中。

其次这三者都是内核态的东西，也就是Linux系统下自动封装，通过C CPP可直接调用api，当内核态数据准备就绪之后，recv会重新进行系统调用，流程和bio一样。

c代码就不贴了，函数具体实现是操作系统提供，给个api用法看看就行

int select(int nfds,fd_set *readset,fd_set *writeset
           ,fd_set *exceptset,const struct timeval *timeout);
nfds：监控的文件描述符集合中最大文件描述符加1。
readset：监控可读事件的文件描述符集合。
writeset：监控可写事件的文件描述符集合。
exceptset：监控异常事件的文件描述符集合。
timeout：超时时间，如果设置为NULL，则select会阻塞直到有文件描述符就绪。

int poll(struct pollfd *fds, nfds_t nfds, int timeout);
fds：这是一个指向pollfd结构数组的指针，
每个pollfd结构都包含了一个文件描述符、一个事件掩码
（指示你对该文件描述符感兴趣的事件类型，如可读、可写、异常）和一个用于返回结果的revents字段。
nfds：这是fds数组中元素的数量。这个值应该大于0。
timeout：这是等待的最长时间（以毫秒为单位）。如果timeout是-1，则poll会无限期地等待，直到至少有一个文件描述符就绪。如果timeout是0，则poll会立即返回，不阻塞。

int epoll_create(int size);
int epoll_ctl(int epollfd, int op, int fd, struct epoll_event *event);
int epoll_wait(int epollfd, struct epoll_event *events, int maxevents, int timeout);
