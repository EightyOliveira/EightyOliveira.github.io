package com.fsms.mybtistest.proxy;

import java.lang.reflect.InvocationHandler;
import java.lang.reflect.Method;

public class MapperProxy implements InvocationHandler {

    @Override
    public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
        System.out.println("method: " + method.getName() + " execute");
        return new MapperMethod().execute();
    }


}

class MapperMethod {
    public Object execute() {
        Object result = null;
        result = "success";
        return result;
    }
}

package com.fsms.mybtistest.proxy;

import java.lang.reflect.Proxy;

public class ProxyDemo {
    public static void main(String[] args) {
        MapperProxy handler = new MapperProxy();

        HelloMapper helloMapper = (HelloMapper) Proxy.newProxyInstance(
                HelloMapper.class.getClassLoader(),
                new Class<?>[]{HelloMapper.class},
                handler);

        System.out.println(helloMapper.sayHello());
    }
}
